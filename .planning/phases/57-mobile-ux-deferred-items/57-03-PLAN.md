```svelte
  import { useBreakpoint } from "$lib/use-breakpoint.svelte";
  const bp = useBreakpoint('lg');
  // ... existing imports & props ...
</script>

{#snippet pickerBody()}

  <BottomSheet open={open} onclose={() => { open = false; }} ariaLabel="<Picker name>">

{/if}
```
