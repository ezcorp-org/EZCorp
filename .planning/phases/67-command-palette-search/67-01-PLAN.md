```typescript
import type { MessageSearchHit } from "$lib/api.js";
import type { Command } from "$lib/command-registry";

// A render-tree node: either a non-clickable header or an actionable row.
export type PaletteRow =
  | { kind: "command"; command: Command }
  | { kind: "hit"; hit: MessageSearchHit };

export type PaletteSection = {
  // "commands" | "in-this-conversation" | "other" | "messages"
  id: string;
  label: string;            // section header text (render-only)
  // For "other"/"messages": project → conversation tree of rows.
  // For "commands"/"in-this-conversation": a flat row list.
  groups: PaletteGroup[];
};

export type PaletteGroup = {
  projectId?: string;       // present for cross-project ("other"/"messages")
  projectName?: string;
  conversationId?: string;
  conversationTitle?: string;
  rows: PaletteRow[];       // actionable rows only
};

export type PaletteResults = {
  sections: PaletteSection[];        // render tree (headers + rows)
  flatItems: (Command | MessageSearchHit)[]; // actionable rows ONLY, in render order
};

// activeConversationId === null when no conversation is active.
export function buildPaletteResults(
  matchingCommands: Command[],
  hits: MessageSearchHit[],
  activeConversationId: string | null,
): PaletteResults;
```
